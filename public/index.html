方式一：<script> 标签（最快上手）

直接在 HTML 中加载 IIFE 包，通过 window.MeowTileKit 访问所有 API。

<script src="http://localhost:3000/dist/index.global.js"></script>
<script>
  const { wgs84ToGcj02, OSMSource, latLngToTile } = window.MeowTileKit;

  const gcj = wgs84ToGcj02(39.9042, 116.4074);
  console.log('GCJ-02:', gcj);

  const url = new OSMSource().getTileUrl({ x: 123, y: 456, z: 13 });
  console.log('瓦片 URL:', url);
</script>

方式二：ES Module

现代浏览器支持 type="module" 直接 import。

<script type="module">
  import { wgs84ToGcj02, AMapSource } from "http://localhost:3000/dist/index.js";

  const gcj = wgs84ToGcj02(39.9042, 116.4074);
  const src = new AMapSource({ style: 'satellite' });
  console.log(src.getTileUrl({ x: 123, y: 456, z: 13 }));
</script>

方式三：npm + 构建工具

生产项目推荐，配合 Vite / Webpack 使用。

npm install meow-tile-kit
import { OSMSource, AMapSource, wgs84ToGcj02, latLngToTile, isInChina } from 'meow-tile-kit';

// 配合 Leaflet
const src = new AMapSource({ style: 'satellite' });
L.tileLayer((c) => src.getTileUrl(c)).addTo(map);

// 坐标转换
const gcj = wgs84ToGcj02(39.9, 116.4);
console.log('在中国?', isInChina(39.9, 116.4));

API 速览

// ── 地图源 ──
new OSMSource()           // OpenStreetMap (WGS-84)
new AMapSource({style})   // 高德地图 (GCJ-02)
new GoogleSource({china}) // Google Maps (WGS-84 / GCJ-02)
new TencentSource()       // 腾讯地图 (GCJ-02)
new BMapSource()          // 百度地图 (BD-09)

// ── 坐标转换 ──
wgs84ToGcj02(lat, lng)   // WGS-84 → GCJ-02
gcj02ToWgs84(lat, lng)   // GCJ-02 → WGS-84
gcj02ToBd09(lat, lng)    // GCJ-02 → BD-09
bd09ToGcj02(lat, lng)    // BD-09 → GCJ-02
wgs84ToBd09(lat, lng)    // WGS-84 → BD-09
isInChina(lat, lng)      // 是否在中国境内
convertCoord(lat, lng, from, to) // 通用转换

// ── 投影 ──
latLngToTile(lat, lng, zoom)     // 经纬度 → 瓦片坐标
tileToLatLng(x, y, z)            // 瓦片坐标 → 经纬度
latLngToPixel(lat, lng, zoom)    // 经纬度 → 像素坐标
